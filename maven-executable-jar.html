How to create executable jar in maven

Mar 30, 2019  |  Ashish Doneriya  |   Software Engineering  

Termiologies

<build>
    <plugins>
      <plugin>
        <artifactId>maven-assembly-plugin</artifactId>
        <configuration>
          <archive>
            <manifest>
              <mainClass>io.github.ashishdoneriya.Test</mainClass>
            </manifest>
          </archive>
          <descriptorRefs>
            <descriptorRef>jar-with-dependencies</descriptorRef>
          </descriptorRefs>
        </configuration>
        <executions>
          <execution>
          
            <!-- this is used for inheritance merges -->
            <id>make-assembly</id>
            
            <!-- bind to the packaging phase --> 
            <phase>package</phase>
            
            <goals>
              <goal>single</goal>
            </goals>
          </execution>
        </executions>
      </plugin>
    </plugins>
  </build>
